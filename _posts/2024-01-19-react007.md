### 기본 세팅


#### index.js

```js
root.render(
  <React.StrictMode>
    <Root />
  </React.StrictMode>
);
```

#### Root.js

```js
const Root = () => {
  return (
    <BrowserRouter>
      <RouterMain/>
    </BrowserRouter>
  );
};
```


#### RouterMain.js

```js
return (
    <div>
        <Routes>
            <Route path="/" element={<PersonList/>} />
            <Route path="/writeform" element={<PersonForm/>} />
            <Route path="/datail/:pnum" element={<PersonDetail/>} />
            <Route path="/updateform/:pnum" element={<PersonUpdateForm/>} />
            
            {/*  위 외 매핑주소일 경우 404 호출  */}
             <Route path='*' element={
                 <div style={{margin:'auto'}}>
                     <img alt="" src = {img404} />
                 </div>
             } />
        </Routes>
    </div>
);
```

const PersonList = () => {
